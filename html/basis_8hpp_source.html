84uint64_t inline GetCPUCycle() {
85#ifdef __x86_64__
86 unsigned int lo, hi;
87 // __asm__ 指示編譯器在此插入彙編語句, 聲明一個內聯彙編表達式
88 // __volatile__ 嚴禁將此處的彙編語句與其它的語句重組合優化
89 // memory
90 // gcc編譯器假設RAM所有記憶體單元均被彙編指令修改，這樣cpu中的registers和cache中已快取的記憶體單元中的數據將作廢,
91 // cpu將不得不在需要的時候重新讀取記憶體中的數據 lfence
92 // 在lfence指令前的讀操作當必須在lfence指令后的讀操作前完成
93 __asm__ __volatile__("lfence" : : : "memory"); // 記憶體屏障
94 // rdtsc: ReaD TimeStamp Counter, it reads a 64-bit counter that counts up at
95 // every clock cycle of your processor.
96 __asm__ __volatile__("rdtsc" : "=a"(lo), "=d"(hi));
97 return ((uint64_t)hi << 32) | lo;
98#elif __aarch64__
99 uint64_t v = 0;
100 // Instruction Synchronization Barrier (ISB) flushes the pipeline in the
101 // processor, so that all instructions following the ISB are fetched from
102 // cache or memory, after the ISB has been completed.
103 asm volatile("isb" : : : "memory");
104 asm volatile(
105 "mrs %0, cntvct_el0"
106 : "=r"(v)); // 虛擬計時器計數暫存器 CNTPCT_EL0物理計時器計數暫存器
107 // asm volatile("mrs %0, pmccntr_el0" : "=r" (v)); // Performance Monitors
108 // Cycle Count Register
109 return v;
110#else
111 printf("unknown arch\n");
112 return 0;
113#endif
114}
